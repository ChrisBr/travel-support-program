def allow_all_comments_to(roles)
  roles = [roles].flatten.map(&:to_sym)
  @roles_for_public_comments ||= []
  @roles_for_private_comments ||= []
  @roles_for_public_comments += roles
  @roles_for_private_comments += roles
end
